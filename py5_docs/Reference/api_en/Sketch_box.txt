@@ meta
name = box()
category = Shape
subcategory = 3D Primitives
type = method
processing_name = box

@@ description
A box is an extruded rectangle. A box with equal dimensions on all sides is a cube.

@@ example
image = Sketch_box_0.png

size(100, 100, P3D)
translate(58, 48, 0)
rotate_y(0.5)
no_fill()
box(40)


@@ example
image = Sketch_box_1.png

size(100, 100, P3D)
translate(58, 48, 0)
rotate_y(0.5)
no_fill()
box(40, 20, 50)
