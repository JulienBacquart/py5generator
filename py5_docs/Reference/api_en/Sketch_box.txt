@@ meta
name = box()
category = Shape
subcategory = 3D Primitives
type = None

@@ description
A box is an extruded rectangle. A box with equal dimensions on all sides is a cube.

@@ example
image = box_0.png

size(100, 100, P3D);
translate(58, 48, 0); 
rotateY(0.5);
noFill();
box(40);

@@ example
image = box_1.png

noFill();
box(40, 20, 50);
