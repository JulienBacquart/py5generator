@@ meta
name = hue()
category = color
subcategory = creating_reading
type = method
pclass = PApplet
processing_name = hue

@@ signatures
hue(rgb: int, /) -> float

@@ variables
rgb: int - any value of the color datatype

@@ description
Extracts the hue value from a color.

@@ example
image = Sketch_hue_0.png

no_stroke()
color_mode(HSB, 255)
c = color(0, 126, 255)
fill(c)
rect(15, 20, 35, 60)
value = hue(c)  # sets 'value' to "0"
fill(value)
rect(50, 20, 35, 60)
