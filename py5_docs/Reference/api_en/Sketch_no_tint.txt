@@ meta
name = no_tint()
category = Image
subcategory = Loading & Displaying
type = None
processing_name = noTint

@@ description
Removes the current fill value for displaying images and reverts to displaying images with their original hues.

@@ example
image = noTint_.png

Py5Image img
img = load_image("laDefense.jpg")
tint(0, 153, 204)  image(img, 0, 0)
no_tint()  image(img, 50, 0)
