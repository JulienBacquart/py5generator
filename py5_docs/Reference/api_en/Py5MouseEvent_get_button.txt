def mouse_clicked(e):
    mouse_button = e.get_button()
    if mouse_button == py5.LEFT:
        py5.println('left mouse click')
    elif mouse_button == py5.CENTER:
        py5.println('center mouse click')
    elif mouse_button == py5.RIGHT:
