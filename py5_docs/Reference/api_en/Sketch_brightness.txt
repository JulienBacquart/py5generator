@@ meta
name = brightness()
category = Color
subcategory = Creating & Reading
type = None
processing_name = brightness

@@ description
Extracts the brightness value from a color.

@@ example
image = brightness_.png

no_stroke()
color_mode(HSB, 255)
color c = color(0, 126, 255)
fill(c)
rect(15, 20, 35, 60)
float value = brightness(c)  fill(value)
rect(50, 20, 35, 60)
