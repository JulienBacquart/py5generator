@@ meta
name = brightness()
category = Color
subcategory = Creating & Reading
type = method
processing_name = brightness

@@ description
Extracts the brightness value from a color.

@@ example
image = Sketch_brightness_0.png

no_stroke()
color_mode(HSB, 255)
c = color(0, 126, 255)
fill(c)
rect(15, 20, 35, 60)
value = brightness(c)  # sets 'value' to 255
fill(value)
rect(50, 20, 35, 60)
