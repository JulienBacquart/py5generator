@@ meta
name = hour()
category = Input
subcategory = Time & Date
type = None
processing_name = hour

@@ description
Processing communicates with the clock on your computer. The ``hour()`` function returns the current hour as a value from 0 - 23.

@@ example
void draw() {
  background(204)
  int s = second()    int m = minute()    int h = hour()      line(s, 0, s, 33)
  line(m, 33, m, 66)
  line(h, 66, h, 100)
}
