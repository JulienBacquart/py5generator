@@ meta
name = end_shape()
category = PShape
subcategory = Method
type = None
processing_name = endShape

@@ description
This method is used to complete a custom shape created with the ``create_shape()`` function. It's always and only used with ``create_shape()``.

@@ example
Py5Shape s  # the Py5Shape object

void setup() {  
  size(100, 100)
  s = create_shape()
  s.begin_shape()
  s.fill(0, 0, 255)
  s.no_stroke()
  s.vertex(0, 0)
  s.vertex(0, 50)
  s.vertex(50, 0)
  s.end_shape()
}

void draw() {
  shape(s, 25, 25)
}
