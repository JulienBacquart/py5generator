@@ meta
name = no_cursor()
category = Environment
subcategory = None
type = None
processing_name = noCursor

@@ description
Hides the cursor from view. Will not work when running the program in a web browser or in full screen (Present) mode.

@@ example
void draw() 
{
  if (mouse_pressed == true) {
    no_cursor()
  } else {
    cursor(HAND)
  }
}
