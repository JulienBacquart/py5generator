@@ meta
name = noCursor()
category = Environment
subcategory = None
type = None

@@ description
Hides the cursor from view. Will not work when running the program in a web browser or in full screen (Present) mode.

@@ example
// Press the mouse to hide the cursor
void draw() 
{
  if (mousePressed == true) {
    noCursor();
  } else {
    cursor(HAND);
  }
}
