@@ meta
name = %py5animatedgif
type = line magic
category = magics
subcategory = line magics

@@ arguments
'filename', type=str
filename of GIF to create

'count', type=int
number of Sketch snapshots to create

'period', type=float
time in seconds between Sketch snapshots

'duration', type=float
time in seconds between frames in the GIF

'-w', '--wait', type=float, dest='wait', default=0.0
wait time in seconds before starting Sketch frame capture

'-l', '--loop', dest='loop', type=int, default=0
number of times for the GIF to loop (default of 0 loops indefinitely)

'--optimize', action='store_true'
optimize GIF palette

@@ description
Create an animated GIF using the currently running Sketch.

Use the ``-w`` argument to wait before starting.

@@ example
py5.run_sketch()
# create a 10 frame animated GIF saved to '/tmp/animated.gif'
%py5animatedgif /tmp/animated.gif 10 1 0.5 -w 3
