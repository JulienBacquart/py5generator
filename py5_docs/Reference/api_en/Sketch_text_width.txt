@@ meta
name = text_width()
category = Typography
subcategory = Metrics
type = None
processing_name = textWidth

@@ description
Calculates and returns the width of any character or text string.

@@ example
image = textWidth_.png

text_size(28)

c = 't'
cw = text_width(c)
text(c, 0, 40)
line(cw, 0, cw, 50) 

s = "Tokyo"
sw = text_width(s)
text(s, 0, 85)
line(sw, 50, sw, 100)
