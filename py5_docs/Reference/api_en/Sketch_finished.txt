@@ description
Boolean variable reflecting if the Sketch has stopped permanently.

@@ example
import time

def draw():
    py5.rect(py5.random_int(py5.width), py5.random_int(py5.height), 10, 10)


py5.run_sketch()
print('sketch has stopped:', py5.finished)
time.sleep(10)

py5.exit_sketch()
print('sketch has stopped:', py5.finished)
