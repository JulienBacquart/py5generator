@@ meta
name = end_record()
category = output
subcategory = files
type = method
pclass = PApplet
processing_name = endRecord

@@ signatures
end_record() -> None

@@ description
Stops the recording process started by ``begin_record()`` and closes the file.

@@ example
def setup():
    size(400, 400)
    begin_record(PDF, "everything.pdf")


def draw():
    ellipse(mouse_x, mouse_y, 10, 10)


def mouse_pressed():
    end_record()
    exit()
