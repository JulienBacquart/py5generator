@@ signatures
end_record() -> None

@@ description
Stops the recording process started by :doc:`begin_record` and closes the file.

@@ example
def settings():
    py5.size(400, 400)


def setup():
    py5.begin_record(py5.PDF, "everything.pdf")


def draw():
    py5.ellipse(py5.mouse_x, py5.mouse_y, 10, 10)


def mouse_pressed():
    py5.end_record()
    py5.exit_sketch()
