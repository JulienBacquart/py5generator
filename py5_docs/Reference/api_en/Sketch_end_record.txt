@@ meta
name = end_record()
category = Output
subcategory = Files
type = None
processing_name = endRecord

@@ description
Stops the recording process started by ``begin_record()`` and closes the file.

@@ example
import processing.pdf.*

void setup() {
  size(400, 400)
  beginRecord(PDF, "everything.pdf")
}

void draw() {
  ellipse(mouseX, mouseY, 10, 10)
}

void mousePressed() {
  endRecord()
  exit()
}
