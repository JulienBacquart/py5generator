@@ meta
name = shader()
category = Rendering
subcategory = Shaders
type = None
processing_name = shader

@@ description
Applies the shader specified by the parameters. It's compatible with the P2D and P3D renderers, but not with the default renderer.

@@ example


def setup():
    global edges
    global img
    size(640, 360, P2D)
    img = load_image("leaves.jpg")
    edges = load_shader("edges.glsl")


def draw():
    shader(edges)
    image(img, 0, 0)
