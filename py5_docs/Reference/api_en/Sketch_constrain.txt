@@ meta
name = constrain()
type = method
category = math
subcategory = calculation

@@ signatures
constrain(amt: Union[float, np.array], low: Union[float, np.array], high: Union[float, np.array]) -> Union[float, np.array]

@@ variables
amt: Union[float, np.array] - the value to constrain
high: Union[float, np.array] - minimum limit
low: Union[float, np.array] - maximum limit

@@ description
Constrains a value to not exceed a maximum and minimum value.

@@ example
def draw():
    py5.background(204)
    mx = py5.constrain(py5.mouse_x, 30, 70)
    py5.rect(mx-10, 40, 20, 20)
