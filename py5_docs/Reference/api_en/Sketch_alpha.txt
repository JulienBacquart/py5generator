@@ meta
name = alpha()
category = color
subcategory = creating_reading
type = method
pclass = PApplet
processing_name = alpha

@@ description
Extracts the alpha value from a color.

@@ example
image = Sketch_alpha_0.png

no_stroke()
c = color(0, 126, 255, 102)
fill(c)
rect(15, 15, 35, 70)
value = alpha(c)  # sets 'value' to 102
fill(value)
rect(50, 15, 35, 70)
