@@ meta
name = alpha()
category = Color
subcategory = Creating & Reading
type = None
processing_name = alpha

@@ description
Extracts the alpha value from a color.

@@ example
image = alpha_.png

no_stroke()
color c = color(0, 126, 255, 102)
fill(c)
rect(15, 15, 35, 70)
float value = alpha(c)  fill(value)
rect(50, 15, 35, 70)
