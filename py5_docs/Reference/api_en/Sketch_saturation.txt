@@ meta
name = saturation()
category = Color
subcategory = Creating & Reading
type = method
pclass = PApplet
processing_name = saturation

@@ description
Extracts the saturation value from a color.

@@ example
image = Sketch_saturation_0.png

no_stroke()
color_mode(HSB, 255)
c = color(0, 126, 255)
fill(c)
rect(15, 20, 35, 60)
value = saturation(c)  # sets 'value' to 126
fill(value)
rect(50, 20, 35, 60)
