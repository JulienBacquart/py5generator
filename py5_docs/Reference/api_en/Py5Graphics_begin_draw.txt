@@ meta
name = begin_draw()
category = PGraphics
subcategory = Method
type = method
pclass = PGraphics
processing_name = beginDraw

@@ signatures
begin_draw() -> None

@@ description
Sets the default properties for a PGraphics object. It should be called before anything is drawn into the object.

@@ example
def settings():
    py5.size(200, 200, py5.P2D)


def setup():
    global pg
    pg = py5.create_graphics(80, 80, py5.P2D)
    pg.begin_draw()
    pg.background(102)
    pg.stroke(255)
    pg.line(20, 20, 80, 80)
    pg.end_draw()


def draw():
    py5.image(pg, 10, 10)
