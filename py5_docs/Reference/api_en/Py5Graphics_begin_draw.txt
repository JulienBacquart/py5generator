@@ meta
name = begin_draw()
category = PGraphics
subcategory = Method
type = None
processing_name = beginDraw

@@ description
Sets the default properties for a PGraphics object. It should be called before anything is drawn into the object.

@@ example
PGraphics pg;

void setup() {
  size(100, 100);
  pg = createGraphics(80, 80, P2D);
  pg.beginDraw();
  pg.background(102);
  pg.stroke(255);
  pg.line(20, 20, 80, 80);
  pg.endDraw();
  noLoop();
}

void draw() {
  image(pg, 10, 10); 
}
