@@ meta
name = texture_wrap()
category = Image
subcategory = Textures
type = None
processing_name = textureWrap

@@ description
Defines if textures repeat or draw once within a texture map. The two parameters are CLAMP (the default behavior) and REPEAT. This function only works with the P2D and P3D renderers.

@@ example


def setup():
    global img
    size(300, 300, P2D)
    img = load_image("berlin-1.jpg")
    texture_mode(NORMAL)


def draw():
    background(0)
    translate(width/2, height/2)
    rotate(map(mouse_x, 0, width, -PI, PI))
    if mouse_pressed:
        texture_wrap(REPEAT)
    else:
        texture_wrap(CLAMP)

    begin_shape()
    texture(img)
    vertex(-100, -100, 0, 0)
    vertex(100, -100, 2, 0)
    vertex(100, 100, 2, 2)
    vertex(-100, 100, 0, 2)
    end_shape()
