@@ meta
name = text_ascent()
category = Typography
subcategory = Metrics
type = None
processing_name = textAscent

@@ description
Returns ascent of the current font at its current size. This information is useful for determining the height of the font above the baseline.

@@ example
image = textAscent_.png

float base = height * 0.75
float scalar = 0.8 
text_size(32)  float a = text_ascent() * scalar  line(0, base-a, width, base-a)
text("dp", 0, base)  
text_size(64)  a = text_ascent() * scalar  line(40, base-a, width, base-a)
text("dp", 40, base)
