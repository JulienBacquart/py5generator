category = PGraphics
subcategory = Method
type = method
pclass = PGraphics
processing_name = endDraw

@@ description
Finalizes the rendering of a PGraphics object so that it can be shown on screen.

@@ example


def setup():
    global pg
    size(100, 100)
    pg = create_graphics(80, 80, P2D)
    pg.begin_draw()
    pg.background(102)
    pg.stroke(255)
    pg.line(20, 20, 80, 80)
    pg.end_draw()
    no_loop()


def draw():
    image(pg, 10, 10)
