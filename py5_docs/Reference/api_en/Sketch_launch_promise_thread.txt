def settings():
    py5.size(250, 100)


def load_data():
    return py5.load_json('http://py5.ixora.io/secret_message.json')


def setup():
    global promise
    promise = py5.launch_promise_thread(load_data)


def draw():
    py5.background(0)
    if promise.is_ready:
        py5.text(promise.result['msg'][:(py5.frame_count // 25)], 20, 50)
