@@ meta
name = no_stroke()
category = color
subcategory = setting
type = method
pclass = PApplet
processing_name = noStroke

@@ description
Disables drawing the stroke (outline). If both ``no_stroke()`` and ``no_fill()`` are called, nothing will be drawn to the screen.

@@ example
image = Sketch_no_stroke_0.png

no_stroke()
rect(30, 20, 55, 55)
