@@ meta
name = reset_shader()
category = Rendering
subcategory = Shaders
type = None
processing_name = resetShader

@@ description
Restores the default shaders. Code that runs after ``reset_shader()`` will not be affected by previously defined shaders.

@@ example
PShader edges  
PImage img
    
void setup() {
  size(640, 360, P2D)
  img = loadImage("leaves.jpg")      
  edges = loadShader("edges.glsl")
}

void draw() {
  shader(edges)
  image(img, 0, 0)
  resetShader()
  image(img, width/2, 0)
}
