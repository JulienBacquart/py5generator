@@ meta
name = loop()
category = Structure
subcategory = None
type = None
processing_name = loop

@@ description
By default, Processing loops through ``draw()`` continuously, executing the code within it. However, the ``draw()`` loop may be stopped by calling ``no_loop()``. In that case, the ``draw()`` loop can be resumed with ``loop()``.

@@ example
void setup() {
  size(200, 200)
  no_loop()  }

float x = 0

void draw() {
  background(204)
  x = x + .1
  if (x > width) {
    x = 0
  }
  line(x, 0, x, height) 
}

void mouse_pressed() {
  loop()  }

void mouse_released() {
  no_loop()  }
