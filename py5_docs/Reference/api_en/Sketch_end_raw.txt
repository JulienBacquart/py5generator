@@ meta
name = end_raw()
category = output
subcategory = files
type = method
pclass = PApplet
processing_name = endRaw

@@ signatures
end_raw() -> None

@@ description
Complement to ``begin_raw()``; they must always be used together. See the ``begin_raw()`` reference for details.

@@ example
def setup():
    size(400, 400)
    begin_raw(PDF, "raw.pdf")


def draw():
    line(pmouse_x, pmouse_y, mouse_x, mouse_y)


def key_pressed():
    if key == ' ':
        end_raw()
        exit()
