@@ meta
name = end_raw()
category = Output
subcategory = Files
type = None
processing_name = endRaw

@@ description
Complement to ``begin_raw()``; they must always be used together. See the ``begin_raw()`` reference for details.

@@ example
import processing.pdf.*;

void setup() {
  size(400, 400);
  beginRaw(PDF, "raw.pdf");
}

void draw() {
  line(pmouseX, pmouseY, mouseX, mouseY);
}

void keyPressed() {
  if (key == ' ') {
    endRaw();
    exit();
  }
}
