@@ meta
name = endRaw()
category = Output
subcategory = Files
type = None

@@ description
Complement to <b>beginRaw()</b>; they must always be used together. See the <b>beginRaw()</b> reference for details.

@@ example
import processing.pdf.*;

void setup() {
  size(400, 400);
  beginRaw(PDF, "raw.pdf");
}

void draw() {
  line(pmouseX, pmouseY, mouseX, mouseY);
}

void keyPressed() {
  if (key == ' ') {
    endRaw();
    exit();
  }
}
