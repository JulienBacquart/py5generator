@@ meta
name = end_raw()
category = output
subcategory = files
type = method
pclass = PApplet
processing_name = endRaw

@@ signatures
end_raw() -> None

@@ description
Complement to ``begin_raw()``; they must always be used together. See the ``begin_raw()`` reference for details.

@@ example
def settings():
    py5.size(400, 400, py5.PDF)


def setup():
    py5.begin_raw(py5.PDF, "raw.pdf")


def draw():
    py5.line(py5.pmouse_x, py5.pmouse_y, py5.mouse_x, py5.mouse_y)


def key_pressed():
    if py5.key == ' ':
        py5.end_raw()
        py5.exit_sketch()
