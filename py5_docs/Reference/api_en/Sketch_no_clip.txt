@@ meta
name = no_clip()
category = Rendering
subcategory = None
type = None
processing_name = noClip

@@ description
Disables the clipping previously started by the ``clip()`` function.

@@ example
def setup():
    size(200, 200)
    image_mode(CENTER)


def draw():
    background(204)
    if mouse_pressed:
        clip(mouse_x, mouse_y, 100, 100)
    else:
        no_clip()

    line(0, 0, width, height)
    line(0, height, width, 0)
